def isEven(int n) -> int {
    if ((n / 2) * 2 == n) {
        return 1
    }
    return 0
}

print(isEven(7));   # Expected Output: 0

# Function that generates an array of multiples of 3
def generateArray(int n) -> int[] {
    if (n <= 0) {
        return []
    }
    int[] arr = [];
    for (int i = 0; i < n; i = i + 1) {
        arr.append(i * 3);
    }
    return arr
}
print(generateArray(5));  # Expected Output: [0, 3, 6, 9, 12]

print("test_no_6");
# Function that doubles each element in the array
def doubleArray(int [] arr) -> int [] {
    for (int i = 0; i < 3; i = i + 1) {
        arr[i] = arr[i] * 2;
    }
    return arr
}
int[] a = [1, 2, 3];
print(doubleArray(a));  # Expected Output: [2, 4, 6]
